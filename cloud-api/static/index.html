🎯 Overview — ESP32-S3 Integration

Gabriel AI mendukung dua mode komunikasi dari firmware ESP32-S3:

📡 REST API
POST Base64-encoded WAV ke /api/process-audio. Cocok untuk firmware awal.
⚡ WebSocket
Stream raw PCM16 audio chunks secara real-time. Latency lebih rendah, lebih efisien.
1 REST API (POST JSON)
// ESP32 Arduino — POST audio ke API #include <HTTPClient.h> #include <ArduinoJson.h> #include <Base64.h> const char* SERVER_URL = "https://YOUR-SERVICE.run.app/api/process-audio"; void sendAudioToGabriel(uint8_t* pcmBuf, size_t len) { // Encode PCM16 as Base64 String b64 = base64::encode(pcmBuf, len); StaticJsonDocument<512> doc; doc["audio"] = b64; doc["session_id"] = "esp32-device-001"; doc["include_audio"] = true; doc["audio_delivery"] = "url"; String payload; serializeJson(doc, payload); HTTPClient http; http.begin(SERVER_URL); http.addHeader("Content-Type", "application/json"); http.addHeader("X-Client", "esp32"); // triggers minimal payload int code = http.POST(payload); if (code == 200) { String resp = http.getString(); // Parse: {"question":"...","answer":"...","audio_url":"/api/audio/..."} DynamicJsonDocument res(1024); deserializeJson(res, resp); Serial.println(res["answer"].as<String>()); } http.end(); }
2 WebSocket — Real-time Audio Stream
// ESP32 — WebSocket binary audio streaming #include <WebSocketsClient.h> #include <ArduinoJson.h> WebSocketsClient ws; const char* WS_HOST = "YOUR-SERVICE.run.app"; const int WS_PORT = 443; void wsEvent(WStype_t type, uint8_t* payload, size_t len) { if (type == WStype_TEXT) { // Parse JSON: {"type":"result","question":"...","answer":"...","audio_url":"..."} DynamicJsonDocument doc(1024); deserializeJson(doc, payload, len); String msgType = doc["type"]; if (msgType == "result") { String answer = doc["answer"]; // Display on OLED, play TTS from audio_url, etc. displayOnOLED(answer); } } } void setup() { ws.beginSSL(WS_HOST, WS_PORT, "/socket.io/?transport=websocket&EIO=4"); ws.onEvent(wsEvent); } void sendAudioStream(uint8_t* pcmBuf, size_t len) { // Send in 5120-byte chunks for (size_t i = 0; i < len; i += 5120) { size_t chunk = min((size_t)5120, len - i); ws.sendBIN(pcmBuf + i, chunk); } // Signal end of audio ws.sendTXT("{\"type\":\"audio_end\"}"); }
3 Audio Format Requirements
Format PCM16 (raw) atau WAV
Sample Rate 16000 Hz
Channels 1 (Mono)
Bit Depth 16-bit
Max Duration 10 detik (320KB)
Mic Gain Server auto-normalize (AGC)
4 WebSocket Message Flow
// CLIENT → SERVER messages (emit 'message' event) { "type": "ping" } // keepalive { "type": "audio_end" } // signal end of audio stream <binary ArrayBuffer> // raw audio chunk (max 5120 bytes) // SERVER → CLIENT messages (on 'message' event) { "type": "connected", "session_id": "..." } // on connect { "type": "pong" } // ping response { "type": "processing" } // STT in progress { "type": "stt_result", "text": "..." } // STT done { "type": "result", "question": "...", // full response "answer": "...", "audio_url": "/api/audio/..." } { "type": "error", "message": "..." } // error
